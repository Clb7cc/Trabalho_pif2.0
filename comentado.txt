// Estrutura de dados para os n√≥s da cobra
struct noparacobra {
  int Xno; //armazena a posi√ß√£o X da cabe√ßa da cobra
  int Yno; //armazena a posi√ß√£o Y da cabe√ßa da cobra
  struct noparacobra *next;
};

// Estrutura de dados para os n√≥s do ranking
struct ranking {
  int score;
  struct ranking *next;
};

// Fun√ß√£o para imprimir informa√ß√µes na parte inferior da tela

  int offsetX = MAXX + 5; //soma 5 ao MAXX, tirando as informa√ß√µes da tela

  screenGotoxy(offsetX, offsetY); //vai para offset X e Y

  screenGotoxy(offsetX, offsetY + 2); //desce + 2 no Y em rela√ß√£o a placar

  screenGotoxy(offsetX, offsetY + 4); //desce + 2 no Y em rela√ß√£o a recorde

// Fun√ß√£o para adicionar um novo n√≥ (parte do corpo) √† cobra
void addcobra(struct noparacobra **head, int x, int y) {
  if (*head == NULL) { //Se score for 0
    *head = (struct noparacobra *)malloc(sizeof(struct noparacobra));
    (*head)->Xno = x; // Define a coordenada X do novo n√≥
    (*head)->Yno = y; // Define a coordenada Y do novo n√≥
    (*head)->next = NULL; // Define proximo como NULL
  } 
  else { // se score n√£o for 0
    struct noparacobra *temp2 = *head; // Cria um ponteiro tempor√°rio apontando para a cabe√ßa da cobra
    struct noparacobra *novo = (struct noparacobra *)malloc(sizeof(struct noparacobra));
    novo->Xno = x;
    novo->Yno = y;
    while (temp2->next != NULL) { //Percorre todas as partes da cobra
      temp2 = temp2->next;
    }
    novo->next = NULL;
    temp2->next = novo; // Adiciona o novo n√≥ ao final da lista
  }
}

// Fun√ß√£o para imprimir a cobra na tela
void printcobra(struct noparacobra *head) {
  struct noparacobra *temp2 = head;
  while (temp2 != NULL) { //enquanto tiver parte do corpo da cobra
    screenSetColor(GREEN, DARKGRAY);
    screenGotoxy(temp2->Xno, temp2->Yno); //coordenada X/Y q tem parte do corpo da cobra
    printf("I");
    temp2 = temp2->next;
  }
}

// Fun√ß√£o para atualizar a posi√ß√£o da cobra na tela
void atualizarcobra(struct noparacobra *head) {
  struct noparacobra *temp2 = head;
  while (temp2 != NULL) {
    screenGotoxy(temp2->Xno, temp2->Yno); //coordenada X/Y q tem parte do corpo da cobra
    printf(" "); //printa um espa√ßo no rabo da cobra quando ela anda
    temp2 = temp2->next;
  }
}

// Fun√ß√£o para liberar a mem√≥ria alocada para os n√≥s da cobra
void freecobra(struct noparacobra **head) {
  struct noparacobra *temp2 = *head;
  while (temp2 != NULL) {
    struct noparacobra *temp1 = temp2; //cria outro temporario
    temp2 = temp2->next; //atualiza a posi√ß√£o do primeiro temporario
    free(temp1); //libera o segundo
  }
}

// Fun√ß√£o para adicionar uma nova cabe√ßa √† cobra
void cobrandando(struct noparacobra **head, int x, int y) {
  struct noparacobra *newHead = (struct noparacobra *)malloc(sizeof(struct noparacobra));
  
  newHead->Xno = x; //nova localiza√ß√£o do X para cabe√ßa da cobra
  newHead->Yno = y; //nova localiza√ß√£o do Y para cabe√ßa da cobra
  newHead->next = *head; //proximo da cabe√ßa passa para a cabe√ßa atual
  *head = newHead; //cabe√ßa recebe localiza√ß√£o da nova cabe√ßa

  struct noparacobra *temp1 = *head;
  while (temp1->next->next != NULL) { //encontra o penultimo no
    temp1 = temp1->next; //faz o penultimo no virar o ultimo
  }

  free(temp1->next); //libera o antigo ultimo no
  temp1->next = NULL;
}

// Fun√ß√£o para imprimir a ma√ß√£ na tela
void printmaca(int x, int y) {
  screenSetColor(RED, DARKGRAY);
  screenGotoxy(x, y);
  printf("O");
}

// Fun√ß√£o para verificar se a cobra colidiu consigo mesma
int baternocorpo(struct noparacobra *head, int x, int y) {
  struct noparacobra *temp2 = head;
  while (temp2 != NULL) {
    if (temp2->Xno == x && temp2->Yno == y) { // Verifica se as coordenadas da cabe√ßa s√£o iguais √†s coordenadas fornecidas como partes do corpo
      return 1; //se sim retorna 1
    }
    temp2 = temp2->next; //adianta para verificar as outras partes do corpo da cobra
  }
  return 0; //se n bater retorna 0
}

// Fun√ß√£o para gerar uma posi√ß√£o aleat√≥ria para a ma√ß√£
void randonmaca(int *x, int *y) {
  *x = rand() % (MAXX - MINX - 1) + MINX + 1; // Gera uma coordenada x aleat√≥ria
  *y = rand() % (MAXY - MINY - 1) + MINY + 1; // Gera uma coordenada y aleat√≥ria
}

// Fun√ß√£o para adicionar uma pontua√ß√£o ao ranking em ordem decrescente
void rankingemordem(struct ranking **head, int score) {
  if (*head == NULL || score > (*head)->score) { // Verifica se a lista est√° vazia ou se a nova pontua√ß√£o √© maior que a primeira
    struct ranking *novo = (struct ranking *)malloc(sizeof(struct ranking)); // Aloca mem√≥ria para um novo n√≥
    novo->score = score; // Define a pontua√ß√£o do novo n√≥
    novo->next = *head; // Aponta o novo n√≥ para a cabe√ßa da lista
    *head = novo; // Atualiza a cabe√ßa da lista
  } else {
    rankingemordem(&((*head)->next), score); // Chama recursivamente para encontrar a posi√ß√£o correta
  }
}

// Fun√ß√£o para adicionar as pontua√ß√µes ao arquivo
void addnoranking(struct ranking *head, FILE *in) {
  struct ranking *temp = head; // Ponteiro tempor√°rio para percorrer a lista
  while (temp != NULL) {
    int score = temp->score; // Obt√©m a pontua√ß√£o do n√≥ atual
    if (fwrite(&score, sizeof(int), 1, in) != 1) { // Escreve a pontua√ß√£o no arquivo
      break; // Interrompe se ocorrer um erro na escrita
    }
    temp = temp->next; // Move para o pr√≥ximo n√≥
  }
}

// Fun√ß√£o para imprimir o ranking na tela
void printranking(struct ranking *head) {
  struct ranking *temp = head; // Ponteiro tempor√°rio para percorrer a lista
  int cont = 0; // Contador para limitar a quantidade de impress√µes
  while (temp != NULL && (cont < 3)) { // Imprime no m√°ximo os tr√™s primeiros colocados
    printf("%d¬∞ colocado: %d pontos\n", cont + 1, temp->score); // Imprime a posi√ß√£o e a pontua√ß√£o
    temp = temp->next; // Move para o pr√≥ximo n√≥
    cont++; // Incrementa o contador
  }
}

// Fun√ß√£o para liberar a mem√≥ria alocada para a lista de ranking
void freeranking(struct ranking **head) {
  struct ranking *temp2 = *head; // Ponteiro tempor√°rio para percorrer a lista
  while (temp2 != NULL) {
    struct ranking *temp1 = temp2; // Ponteiro tempor√°rio para armazenar o n√≥ atual
    temp2 = temp2->next; // Move para o pr√≥ximo n√≥
    free(temp1); // Libera a mem√≥ria do n√≥ atual
  }
}

// Loop principal do jogo
void jogoLoop(struct noparacobra *head, int *dirX, int *dirY, int *placar, int *recorde, time_t tempoinicial, int PosMacaX, int PosMacaY) {
  int ch = 0; // Vari√°vel para armazenar a entrada do usu√°rio
  while (ch != KEY_ESC) { // Continua o loop at√© o usu√°rio pressionar a tecla de escape
    if (keyhit()) { // Verifica se uma tecla foi pressionada
      ch = readch(); // L√™ a tecla pressionada
      switch (ch) {
        case KEY_W: // Se a tecla 'W' foi pressionada
          if (*dirY != 1) { // Verifica se a dire√ß√£o n√£o √© para baixo
            *dirX = 0; // Define a dire√ß√£o x como zero
            *dirY = -1; // Define a dire√ß√£o y como para cima
          }
          break;
        case KEY_S: // Se a tecla 'S' foi pressionada
          if (*dirY != -1) { // Verifica se a dire√ß√£o n√£o √© para cima
            *dirX = 0; // Define a dire√ß√£o x como zero
            *dirY = 1; // Define a dire√ß√£o y como para baixo
          }
          break;
        case KEY_A: // Se a tecla 'A' foi pressionada
          if (*dirX != 1) { // Verifica se a dire√ß√£o n√£o √© para a direita
            *dirX = -1; // Define a dire√ß√£o x como para a esquerda
            *dirY = 0; // Define a dire√ß√£o y como zero
          }
          break;
        case KEY_D: // Se a tecla 'D' foi pressionada
          if (*dirX != -1) { // Verifica se a dire√ß√£o n√£o √© para a esquerda
            *dirX = 1; // Define a dire√ß√£o x como para a direita
            *dirY = 0; // Define a dire√ß√£o y como zero
          }
          break;
      }
      screenUpdate(); // Atualiza a tela
    }
    if (timerTimeOver() == 1) { // Verifica se o temporizador expirou
      int newX = head->Xno + *dirX; // Calcula a nova posi√ß√£o x da cabe√ßa da cobra
      int newY = head->Yno + *dirY; // Calcula a nova posi√ß√£o y da cabe√ßa da cobra
      if (newX >= (MAXX) || newX <= MINX || newY >= MAXY || newY <= MINY) { // Verifica se a nova posi√ß√£o est√° fora dos limites
        break; // Sai do loop se estiver fora dos limites
      }
      if (baternocorpo(head, newX, newY) == 1) { // Verifica se a cobra colidiu com seu pr√≥prio corpo
        break; // Sai do loop se houve colis√£o
      }
      if (newX == PosMacaX && newY == PosMacaY) { // Verifica se a cobra comeu a ma√ß√£
        addcobra(&head, PosMacaX, PosMacaY); // Adiciona um segmento √† cobra
        randonmaca(&PosMacaX, &PosMacaY); // Gera uma nova posi√ß√£o para a ma√ß√£
        printmaca(PosMacaX, PosMacaY); // Imprime a nova ma√ß√£ na tela
        (*placar)++; // Incrementa a pontua√ß√£o
      }
      atualizarcobra(head); // Atualiza a posi√ß√£o da cobra
      cobrandando(&head, newX, newY); // Move a cabe√ßa da cobra para a nova posi√ß√£o
      printcobra(head); // Imprime a cobra na tela
      screenUpdate(); // Atualiza a tela
      printembaixo(*placar, *recorde, (int)difftime(time(NULL), tempoinicial)); // Imprime a pontua√ß√£o, recorde e tempo decorrido
    }
  }
}
